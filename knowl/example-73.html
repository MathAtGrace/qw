\begin{align*} \bar{x}\amp=\frac{(15)(6)+(20)(8)+(25)(11)+(30)(17)+(35)(19)+(40)(20)+(45)(12)+(50)(7)+(5000)(1)}{100}\\ \amp=\frac{8390}{101}\\ \amp=83.069 \end{align*}
